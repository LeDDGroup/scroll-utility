import { Playground } from 'docz'
import ScrollUtility from 'scroll-utility'
import { Ruler } from './Ruler'

# Playground

<Playground>
	{() => {
		const ScrollManager = React.useMemo(() => new ScrollUtility('#container'), []);
		const scrollBottom = () => ScrollManager.scrollTop += 100
		const getScrollTop = () => alert(ScrollManager.scrollTop);
		return (
			<>
				<Ruler />
				<button onClick={scrollBottom}>Scroll Bottom</button>
				<button onClick={getScrollTop}>Get scroll position </button>
				<div id='container' style={{ height: '50vh', overflow: 'auto'}}><div style={{height: '100vh'}} /></div>
			</>
		)
	}}
</Playground>
